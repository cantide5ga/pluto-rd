"use strict";
var WordCloudActionCreator_1 = require('../flux/actions/WordCloudActionCreator');
var BlogActionCreator_1 = require('../flux/actions/BlogActionCreator');
// data exchange logic
var DataSourceConnectorStatic = (function () {
    function DataSourceConnectorStatic() {
        /**
         * Registers a callback driver for accessing entries
         *
         * @param keyword The keyword that has been clicked.
         */
        this.registerEntryDriver = function (queryFn) {
            BlogActionCreator_1.BlogActionCreator.initGetEntries(queryFn);
        };
        this.registerKeywordDriver = function (keywords, entryCount) {
            var data = {
                keywords: keywords,
                entryCount: entryCount
            };
            WordCloudActionCreator_1.WordCloudActionCreator.initKeywords(data);
        };
    }
    return DataSourceConnectorStatic;
}());
exports.Ds = new DataSourceConnectorStatic();
